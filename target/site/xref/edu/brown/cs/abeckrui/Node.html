3   /**
4    * This node class contains CordComparable objects that can be used to build the KDTree.
5    * It holds CordComparable objects so that it can be more extensible. All that is required
6    * is that the objects are able to be compared coordinate wise and has some number of
7    * dimensions.
8    */
9   public class Node {
10  
11    private Node left;
12    private Node right;
13    //this is the comparable object the Node will store
14    private CordComparable compObject;
15  
16    /**
17     * In the constructor, I set the children initially to null. I also store the CordComparable
18     * object passed in as an instance variable.
19     * @param comparableObject representing CordComparable object to store in node.
20     */
21    public Node(CordComparable comparableObject) {
22      left = null;
23      right = null;
24      compObject = comparableObject;
25    }
26  
27    /**
28     * Returns left child.
29     * @return Node representing left child.
30     */
31    public Node getLeft() {
32      return left;
33    }
34  
35    /**
36     * Checks if node has left child.
37     * @return true if node has left child.
38     */
39    public boolean hasLeft() {
40      if (left == null) {
41        return false;
42      }
43      return true;
44    }
45  
46    /**
47     * Returns right child.
48     * @return Node representing right child.
49     */
50    public Node getRight() {
51      return right;
52    }
53  
54    /**
55     * Checks if node has right child.
56     * @return true if node has right child.
57     */
58    public boolean hasRight() {
59      if (right == null) {
60        return false;
61      }
62      return true;
63    }
64  
65    /**
66     * Sets left child.
67     * @param leftChild node to be set as child.
68     */
69    public void setLeft(Node leftChild) {
70      if (left != null) {
71        System.err.println("ERROR: Left child is not null!");
72      }
73      left = leftChild;
74    }
75  
76    /**
77     * Sets right child.
78     * @param rightChild node to be set as child.
79     */
80    public void setRight(Node rightChild) {
81      right = rightChild;
82    }
83  
84    /**
85     * Getter method to return the object stored in the Node.
86     * @return CordComprable object contained within the Node
87     */
88    public CordComparable getCompObject() {
89      return compObject;
90    }
91  }
