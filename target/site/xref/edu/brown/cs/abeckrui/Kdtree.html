3   import java.util.Collections;
4   import java.util.List;
5   
6   public class Kdtree {
7   
8     private Node root;
9     private List<Node> nodes;
10  
11    public Kdtree(List<Node> nodesList){
12      nodes = nodesList;
13      root = this.buildTree(nodes, 0);
14    }
15  
16    /**
17     * This function builds the KD tree recursively given a list of nodes
18     * @return Node representing root node of current recursion
19     */
20    private Node buildTree(List<Node> nodes, int depth) {
21      //edge case if list is empty - this means the root will be null, or wont exist
22      if (nodes.size() == 0){
23        return null;
24      }
25      //base case if list is one, just return this as root node
26      if (nodes.size() == 1){
27        return nodes.get(0);
28      }
29      Collections.sort(nodes, new CoordinateComparator(depth));
30      int medianIndex = (nodes.size() - 1) / 2;
31      //variable to compare median coordinates with duplicates
32      double medianCoordinate = nodes.get(medianIndex).getCompObject().getCoordinate(depth);
33      //want to get furthest left median (if there are multiple objects with same value)
34      while (medianIndex > 0) {
35        if (nodes.get(medianIndex - 1).getCompObject().getCoordinate(depth) != medianCoordinate) {
36          break;
37        }
38        medianIndex--;
39      }
40      Node median = nodes.get(medianIndex);
41      List<Node> leftList = nodes.subList(0,medianIndex);
42      List<Node> rightList = nodes.subList(medianIndex + 1, nodes.size());
43      //recurse to set left and right children
44      median.setLeft(this.buildTree(leftList, depth + 1));
45      median.setRight(this.buildTree(rightList, depth + 1));
46      return median;
47    }
48  
49    /**
50     *
51     * @return Node representing the root node
52     */
53    public Node getRoot(){
54      return root;
55    }
56  
57    /**
58     *
59     * @return a list of nodes representing the Nodes in the KDtree
60     */
61    public List<Node> getNodeList(){
62      return nodes;
63    }
64  }
